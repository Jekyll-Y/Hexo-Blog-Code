~~~c++
namespace IO
{
    inline int read()
    {
        int x = 0, f = 1;
        char ch = getchar();
        while(!isdigit(ch))
        {
            if(ch == '-')
                f = -1;
            ch = getchar();
        }
        while(isdigit(ch))
        {
            x =((x << 3) + (x << 1)) + ch - '0';
            ch = getchar();
        }
        return x * f;
    }
    inline void write(int x)
    {
        char ch[100];
        int len = 0;
        ch[++len] = '\n';
        if(x < 0)
            putchar('-'), x = -x;
        while(x != 0)
        {
            ch[++len] = x % 10 + '0';
            x /= 10;
        }
        while(len)
            putchar(ch[len--]);
    }
    inline long long readL()
    {
        long long x = 0, f = 1;
        char ch = getchar();
        while(!isdigit(ch))
        {
            if(ch == '-')
                f = -1;
            ch = getchar();
        }
        while(isdigit(ch))
        {
            x =((x << 3ll) + (x << 1ll)) + ch - '0';
            ch = getchar();
        }
        return x * f;
    }
    inline void writeL(long long x)
    {
        char ch[100];
        int len = 0;
        ch[++len] = '\n';
        if(x < 0)
            putchar('-'), x = -x;
        while(x != 0)
        {
            ch[++len] = x % 10ll + '0';
            x /= 10ll;
        }
        while(len)
            putchar(ch[len--]);
    }
    inline string readS()
    {
        string str = "";
        char ch = getchar();
        while(ch == ' ' || ch == '\n')
            ch = getchar();
        while(ch != ' ' && ch != '\n')
        {
            str += ch;
            ch = getchar();
        }
        return str;
    }
    inline void writeS(string s)
    {
        for(int i = 0; i < s.size(); i++)
            putchar(s[i]);
        putchar('\n');
    }
    inline double readD()
    {
        double x = 0, f = 1, k = 0.1;
        char ch = getchar();
        while(ch == ' ' || ch == '\n' || ch == '-')
        {
            if(ch == '-')
                f = -1;
            ch = getchar();
        }
        while(isdigit(ch))
        {
            x = x * 10 + ch - '0';
            ch = getchar();
        }
        if(ch == '.')
        {
            ch = getchar();
            while(isdigit(ch))
            {
                x = x + 1.0 *(ch - '0') * k;
                k *= 0.1;
                ch = getchar();
            }
        }
        return x * f;
    }
    inline void writeD(double x)
    {
        printf("%.8lf\n", x);
    }
}

using namespace IO;
~~~
